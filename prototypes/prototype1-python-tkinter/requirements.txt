# Gertrude Shell - Python Dependencies
# ======================================
# Install with:  pip install -r requirements.txt
#
# Core requirement: Python 3.8+
# tkinter is bundled with CPython - no separate install needed.
#
# On Windows, tkinter ships with the standard Python installer.
# If missing on Linux: sudo apt install python3-tk

# Pillow is used to display real photos in the photo placeholders.
# If Pillow is not installed, the app falls back to gray canvas placeholders.
# The app will run normally without Pillow - it is optional but recommended.
Pillow>=9.0.0

# pytest is only needed for running the unit tests (tests/test_logic.py).
# It is not required to run gertrude_shell.py itself.
pytest>=7.0.0
